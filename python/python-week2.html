1. 條件判斷：if 語句

基本 if 語句

# 基本 if 語句
age = 20

if age >= 18:
    print("你已成年")

# 輸出：你已成年

if-else 語句

# if-else 雙向判斷
temperature = 25

if temperature > 30:
    print("天氣很熱")
else:
    print("天氣還好")

# 輸出：天氣還好

if-elif-else 多重判斷

# 成績等級判定
score = 85

if score >= 90:
    grade = "A"
elif score >= 80:
    grade = "B"
elif score >= 70:
    grade = "C"
elif score >= 60:
    grade = "D"
else:
    grade = "F"

print(f"成績等級：{grade}")
# 輸出：成績等級：B
重點提醒：
  • 條件判斷後面必須加冒號 :
  • 程式碼區塊必須縮排（建議 4 個空格）
  • elif 可以有多個，else 最多一個且必須在最後
  • 條件由上而下依序判斷，符合就執行並跳出

複合條件判斷

# 使用 and（且）
age = 20
has_license = True

if age >= 18 and has_license:
    print("可以開車")


# 使用 or（或）
is_weekend = True
is_holiday = False

if is_weekend or is_holiday:
    print("可以休息")


# 使用 not（非）
is_raining = False

if not is_raining:
    print("可以出門運動")


# 組合使用
score = 75
attendance = 90

if (score >= 60 and attendance >= 80) or score >= 90:
    print("通過考核")

巢狀條件判斷

# 判斷是否符合申請資格
age = 25
income = 50000
credit_score = 700

if age >= 20:
    print("年齡符合")
    
    if income >= 30000:
        print("收入符合")
        
        if credit_score >= 650:
            print("信用評分符合")
            print("核准申請！")
        else:
            print("信用評分不足")
    else:
        print("收入不足")
else:
    print("年齡不符")
巢狀判斷建議：

巢狀太深會降低可讀性，建議使用複合條件（and/or）簡化：

# 簡化版本（推薦）
if age >= 20 and income >= 30000 and credit_score >= 650:
    print("核准申請！")
else:
    print("申請條件不符")

三元運算符（條件表達式）

# 傳統寫法
age = 20
if age >= 18:
    status = "成年"
else:
    status = "未成年"


# 三元運算符（簡化寫法）
status = "成年" if age >= 18 else "未成年"


# 實用範例
score = 85
result = "及格" if score >= 60 else "不及格"
print(result)  # 輸出：及格


# 搭配函數使用
numbers = [1, 2, 3, 4, 5]
max_num = max(numbers) if numbers else 0

2. for 迴圈

基本 for 迴圈

# 遍歷串列
fruits = ["蘋果", "香蕉", "橘子"]

for fruit in fruits:
    print(fruit)

# 輸出：
# 蘋果
# 香蕉
# 橘子


# 遍歷字串
text = "Python"

for char in text:
    print(char)

# 輸出：P y t h o n（每個字元一行）

range() 函數

# range(結束值) - 從 0 開始
for i in range(5):
    print(i)
# 輸出：0 1 2 3 4


# range(起始值, 結束值)
for i in range(1, 6):
    print(i)
# 輸出：1 2 3 4 5


# range(起始值, 結束值, 步進值)
for i in range(0, 10, 2):
    print(i)
# 輸出：0 2 4 6 8


# 倒數計數
for i in range(5, 0, -1):
    print(i)
# 輸出：5 4 3 2 1

enumerate() 函數

# 同時取得索引和值
fruits = ["蘋果", "香蕉", "橘子"]

for index, fruit in enumerate(fruits):
    print(f"{index}: {fruit}")

# 輸出：
# 0: 蘋果
# 1: 香蕉
# 2: 橘子


# 指定起始索引
for index, fruit in enumerate(fruits, 1):
    print(f"{index}. {fruit}")

# 輸出：
# 1. 蘋果
# 2. 香蕉
# 3. 橘子

遍歷字典

student = {
    "name": "Alice",
    "age": 20,
    "grade": "A"
}

# 遍歷鍵
for key in student:
    print(key)


# 遍歷值
for value in student.values():
    print(value)


# 同時遍歷鍵和值
for key, value in student.items():
    print(f"{key}: {value}")

# 輸出：
# name: Alice
# age: 20
# grade: A

3. while 迴圈

基本 while 迴圈

# 基本計數
count = 1

while count <= 5:
    print(f"計數：{count}")
    count += 1

# 輸出：
# 計數：1
# 計數：2
# 計數：3
# 計數：4
# 計數：5

使用者輸入驗證

# 持續要求輸入直到正確
password = ""

while password != "1234":
    password = input("請輸入密碼：")
    
    if password != "1234":
        print("密碼錯誤，請重新輸入")

print("登入成功！")

限制嘗試次數

# 最多嘗試 3 次
attempts = 0
max_attempts = 3
correct_password = "1234"

while attempts < max_attempts:
    password = input("請輸入密碼：")
    
    if password == correct_password:
        print("登入成功！")
        break
    else:
        attempts += 1
        remaining = max_attempts - attempts
        print(f"密碼錯誤，還有 {remaining} 次機會")
else:
    print("嘗試次數已用完，帳號已鎖定")

無窮迴圈（配合 break）

# 選單系統
while True:
    print("\n=== 選單 ===")
    print("1. 查詢資料")
    print("2. 新增資料")
    print("3. 離開")
    
    choice = input("請選擇功能 (1-3)：")
    
    if choice == "1":
        print("執行查詢...")
    elif choice == "2":
        print("執行新增...")
    elif choice == "3":
        print("感謝使用，再見！")
        break
    else:
        print("無效的選擇，請重新輸入")
注意無窮迴圈：

使用 while True 時，務必確保有適當的 break 條件，否則程式會無限執行。

4. break 與 continue

break：跳出迴圈

# 找到目標就停止
numbers = [1, 3, 5, 7, 9, 11]
target = 7

for num in numbers:
    if num == target:
        print(f"找到目標：{num}")
        break
    print(f"檢查：{num}")

# 輸出：
# 檢查：1
# 檢查：3
# 檢查：5
# 找到目標：7

continue：跳過本次迭代

# 只印出奇數
for i in range(1, 11):
    if i % 2 == 0:
        continue  # 跳過偶數
    print(i)

# 輸出：1 3 5 7 9


# 過濾負數
numbers = [10, -5, 8, -3, 15, -1]
total = 0

for num in numbers:
    if num < 0:
        continue  # 跳過負數
    total += num

print(f"正數總和：{total}")  # 33

else 子句

# for-else：迴圈正常結束時執行
numbers = [1, 3, 5, 7, 9]
target = 6

for num in numbers:
    if num == target:
        print("找到了！")
        break
else:
    print("沒有找到目標")

# 輸出：沒有找到目標
else 子句說明：
  • 只有在迴圈正常結束時才會執行
  • 如果透過 break 跳出，則 else 不會執行
  • 常用於搜尋操作，判斷是否找到目標

5. 巢狀迴圈

基本巢狀迴圈

# 印出矩形圖案
for i in range(3):
    for j in range(5):
        print("*", end="")
    print()  # 換行

# 輸出：
# *****
# *****
# *****

九九乘法表

# 完整九九乘法表
print("九九乘法表")
print("=" * 60)

for i in range(1, 10):
    for j in range(1, 10):
        result = i * j
        print(f"{i} × {j} = {result:2}", end="  ")
    print()

三角形圖案

# 直角三角形
for i in range(1, 6):
    for j in range(i):
        print("*", end="")
    print()

# 輸出：
# *
# **
# ***
# ****
# *****


# 等腰三角形
n = 5
for i in range(1, n + 1):
    # 印空格
    for j in range(n - i):
        print(" ", end="")
    # 印星號
    for j in range(2 * i - 1):
        print("*", end="")
    print()

# 輸出：
#     *
#    ***
#   *****
#  *******
# *********

6. 實用範例

範例一：猜數字遊戲

import random

# 產生 1-100 的隨機數字
answer = random.randint(1, 100)
attempts = 0
max_attempts = 7

print("=== 猜數字遊戲 ===")
print(f"我想了一個 1-100 的數字，你有 {max_attempts} 次機會")

while attempts < max_attempts:
    guess = int(input(f"\n第 {attempts + 1} 次猜測，請輸入數字："))
    attempts += 1
    
    if guess == answer:
        print(f"恭喜你！猜對了！答案是 {answer}")
        print(f"你總共猜了 {attempts} 次")
        break
    elif guess < answer:
        print("太小了！")
    else:
        print("太大了！")
    
    remaining = max_attempts - attempts
    if remaining > 0:
        print(f"還有 {remaining} 次機會")
else:
    print(f"\n很遺憾，機會用完了！答案是 {answer}")

範例二：質數判斷

# 判斷一個數是否為質數
num = int(input("請輸入一個正整數："))

if num <= 1:
    print(f"{num} 不是質數")
else:
    is_prime = True
    
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            is_prime = False
            break
    
    if is_prime:
        print(f"{num} 是質數")
    else:
        print(f"{num} 不是質數")

範例三：階乘計算

# 計算 n!
n = int(input("請輸入一個正整數："))
factorial = 1

for i in range(1, n + 1):
    factorial *= i

print(f"{n}! = {factorial}")
