1. Web 開發基礎概念

HTTP 協定基礎

HTTP 請求方法：

  • GET：取得資源（查詢資料）
  • POST：提交資料（新增資料）
  • PUT：更新資源（完整更新）
  • PATCH：部分更新資源
  • DELETE：刪除資源

HTTP 狀態碼

狀態碼 說明 範例
2xx 成功 200 OK, 201 Created
3xx 重新導向 301 Moved Permanently, 302 Found
4xx 客戶端錯誤 400 Bad Request, 404 Not Found
5xx 伺服器錯誤 500 Internal Server Error

MVC 架構

Model-View-Controller 模式：

  • Model（模型）：資料與業務邏輯
  • View（視圖）：使用者介面（HTML 模板）
  • Controller（控制器）：處理請求，協調 Model 與 View

為什麼選擇 Flask？

Flask 的特點：

  • ✓ 輕量級：核心簡單，容易學習
  • ✓ 靈活：不強制特定架構
  • ✓ 擴展性好：豐富的擴展套件
  • ✓ 適合小型專案：快速開發原型
  • ✓ 文件完整：社群活躍

表單驗證

# app.py
from flask import Flask, render_template, request, flash, redirect, url_for

app = Flask(__name__)
app.secret_key = 'your-secret-key-here'  # 用於 flash 訊息

@app.route('/')
def index():
    return render_template('form.html')

@app.route('/submit', methods=['POST'])
def submit():
    # 取得表單資料
    name = request.form.get('name', '').strip()
    email = request.form.get('email', '').strip()
    age = request.form.get('age', '')
    
    # 驗證
    errors = []
    
    if not name:
        errors.append('姓名為必填')
    elif len(name) < 2:
        errors.append('姓名至少 2 個字元')
    
    if not email:
        errors.append('Email 為必填')
    elif '@' not in email:
        errors.append('Email 格式錯誤')
    
    if age:
        try:
            age = int(age)
            if age < 0 or age > 150:
                errors.append('年齡必須在 0-150 之間')
        except ValueError:
            errors.append('年齡必須是數字')
    
    # 如果有錯誤，返回表單
    if errors:
        for error in errors:
            flash(error, 'error')
        return redirect(url_for('index'))
    
    # 驗證通過，處理資料
    flash('表單提交成功！', 'success')
    return render_template('success.html', name=name)


if __name__ == '__main__':
    app.run(debug=True)

檔案上傳

# app.py
from flask import Flask, render_template, request, flash, redirect, url_for
from werkzeug.utils import secure_filename
import os

app = Flask(__name__)
app.secret_key = 'your-secret-key'

# 設定上傳資料夾
UPLOAD_FOLDER = 'uploads'
ALLOWED_EXTENSIONS = {'png', 'jpg', 'jpeg', 'gif', 'pdf'}

app.config['UPLOAD_FOLDER'] = UPLOAD_FOLDER
app.config['MAX_CONTENT_LENGTH'] = 16 * 1024 * 1024  # 限制 16MB

# 建立上傳資料夾
os.makedirs(UPLOAD_FOLDER, exist_ok=True)


def allowed_file(filename):
    """檢查檔案類型"""
    return '.' in filename and \
           filename.rsplit('.', 1)[1].lower() in ALLOWED_EXTENSIONS


@app.route('/')
def index():
    return render_template('upload.html')


@app.route('/upload', methods=['POST'])
def upload():
    # 檢查是否有檔案
    if 'file' not in request.files:
        flash('沒有選擇檔案', 'error')
        return redirect(url_for('index'))
    
    file = request.files['file']
    
    # 檢查檔案名稱
    if file.filename == '':
        flash('沒有選擇檔案', 'error')
        return redirect(url_for('index'))
    
    # 檢查檔案類型並儲存
    if file and allowed_file(file.filename):
        filename = secure_filename(file.filename)
        filepath = os.path.join(app.config['UPLOAD_FOLDER'], filename)
        file.save(filepath)
        
        flash(f'檔案上傳成功：{filename}', 'success')
        return render_template('success.html', filename=filename)
    else:
        flash('不允許的檔案類型', 'error')
        return redirect(url_for('index'))


if __name__ == '__main__':
    app.run(debug=True)

7. Session 與 Cookie

Session 基本使用

# app.py
from flask import Flask, render_template, request, session, redirect, url_for

app = Flask(__name__)
app.secret_key = 'your-secret-key-here'  # Session 需要 secret_key


@app.route('/')
def index():
    # 取得 session 中的資料
    username = session.get('username')
    return render_template('index.html', username=username)


@app.route('/login', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        username = request.form.get('username')
        password = request.form.get('password')
        
        # 簡單驗證（實際應用應該檢查資料庫）
        if username == 'admin' and password == 'password':
            # 儲存到 session
            session['username'] = username
            session['logged_in'] = True
            return redirect(url_for('dashboard'))
        else:
            return render_template('login.html', error='Invalid credentials')
    
    return render_template('login.html')


@app.route('/dashboard')
def dashboard():
    # 檢查是否登入
    if not session.get('logged_in'):
        return redirect(url_for('login'))
    
    return render_template('dashboard.html', username=session.get('username'))


@app.route('/logout')
def logout():
    # 清除 session
    session.clear()
    return redirect(url_for('index'))


if __name__ == '__main__':
    app.run(debug=True)

Cookie 操作

# app.py
from flask import Flask, request, make_response, render_template
from datetime import datetime, timedelta

app = Flask(__name__)


@app.route('/')
def index():
    # 讀取 Cookie
    visits = request.cookies.get('visits', 0)
    visits = int(visits) + 1
    
    # 建立 response 並設定 Cookie
    response = make_response(render_template('index.html', visits=visits))
    
    # 設定 Cookie（過期時間：30 天）
    expires = datetime.now() + timedelta(days=30)
    response.set_cookie('visits', str(visits), expires=expires)
    
    return response


@app.route('/preferences', methods=['GET', 'POST'])
def preferences():
    if request.method == 'POST':
        theme = request.form.get('theme', 'light')
        language = request.form.get('language', 'en')
        
        response = make_response(redirect(url_for('index')))
        response.set_cookie('theme', theme, max_age=60*60*24*365)  # 1 年
        response.set_cookie('language', language, max_age=60*60*24*365)
        
        return response
    
    # 讀取當前偏好
    theme = request.cookies.get('theme', 'light')
    language = request.cookies.get('language', 'en')
    
    return render_template('preferences.html', theme=theme, language=language)


@app.route('/clear-cookies')
def clear_cookies():
    """清除所有 Cookies"""
    response = make_response(redirect(url_for('index')))
    
    # 刪除 Cookie（設定過期時間為過去）
    response.set_cookie('visits', '', expires=0)
    response.set_cookie('theme', '', expires=0)
    response.set_cookie('language', '', expires=0)
    
    return response


if __name__ == '__main__':
    app.run(debug=True)

購物車範例（使用 Session）

# app.py
from flask import Flask, render_template, request, session, redirect, url_for, flash

app = Flask(__name__)
app.secret_key = 'your-secret-key'

# 模擬商品資料
PRODUCTS = {
    1: {'name': 'Python Book', 'price': 500},
    2: {'name': 'Flask Book', 'price': 450},
    3: {'name': 'Django Book', 'price': 480},
}


@app.route('/')
def index():
    return render_template('products.html', products=PRODUCTS)


@app.route('/add-to-cart/')
def add_to_cart(product_id):
    """加入購物車"""
    if product_id not in PRODUCTS:
        flash('商品不存在', 'error')
        return redirect(url_for('index'))
    
    # 初始化購物車
    if 'cart' not in session:
        session['cart'] = {}
    
    # 加入購物車
    cart = session['cart']
    product_id_str = str(product_id)
    
    if product_id_str in cart:
        cart[product_id_str] += 1
    else:
        cart[product_id_str] = 1
    
    session['cart'] = cart  # 更新 session
    session.modified = True  # 標記 session 已修改
    
    flash('已加入購物車', 'success')
    return redirect(url_for('index'))


@app.route('/cart')
def cart():
    """檢視購物車"""
    cart = session.get('cart', {})
    
    # 計算總價
    items = []
    total = 0
    
    for product_id_str, quantity in cart.items():
        product_id = int(product_id_str)
        if product_id in PRODUCTS:
            product = PRODUCTS[product_id]
            subtotal = product['price'] * quantity
            total += subtotal
            
            items.append({
                'id': product_id,
                'name': product['name'],
                'price': product['price'],
                'quantity': quantity,
                'subtotal': subtotal
            })
    
    return render_template('cart.html', items=items, total=total)


@app.route('/remove-from-cart/')
def remove_from_cart(product_id):
    """從購物車移除"""
    cart = session.get('cart', {})
    product_id_str = str(product_id)
    
    if product_id_str in cart:
        del cart[product_id_str]
        session['cart'] = cart
        session.modified = True
        flash('已從購物車移除', 'success')
    
    return redirect(url_for('cart'))


@app.route('/clear-cart')
def clear_cart():
    """清空購物車"""
    session.pop('cart', None)
    flash('購物車已清空', 'success')
    return redirect(url_for('cart'))


if __name__ == '__main__':
    app.run(debug=True)

# app.py
from flask import Flask, render_template, request, redirect, url_for, flash
import sqlite3

app = Flask(__name__)
app.secret_key = 'your-secret-key'

DATABASE = 'blog.db'


def get_db():
    """取得資料庫連線"""
    conn = sqlite3.connect(DATABASE)
    conn.row_factory = sqlite3.Row  # 讓查詢結果可以用字典方式存取
    return conn


def init_db():
    """初始化資料庫"""
    conn = get_db()
    cursor = conn.cursor()
    
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS posts (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            title TEXT NOT NULL,
            content TEXT NOT NULL,
            author TEXT NOT NULL,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
    ''')
    
    conn.commit()
    conn.close()


# 在應用啟動時初始化資料庫
with app.app_context():
    init_db()


@app.route('/')
def index():
    """顯示所有文章"""
    conn = get_db()
    cursor = conn.cursor()
    
    cursor.execute('SELECT * FROM posts ORDER BY created_at DESC')
    posts = cursor.fetchall()
    
    conn.close()
    
    return render_template('posts.html', posts=posts)


@app.route('/post/')
def post_detail(post_id):
    """顯示文章詳情"""
    conn = get_db()
    cursor = conn.cursor()
    
    cursor.execute('SELECT * FROM posts WHERE id = ?', (post_id,))
    post = cursor.fetchone()
    
    conn.close()
    
    if not post:
        flash('文章不存在', 'error')
        return redirect(url_for('index'))
    
    return render_template('post_detail.html', post=post)


@app.route('/new', methods=['GET', 'POST'])
def new_post():
    """新增文章"""
    if request.method == 'POST':
        title = request.form.get('title', '').strip()
        content = request.form.get('content', '').strip()
        author = request.form.get('author', '').strip()
        
        # 驗證
        if not title or not content or not author:
            flash('所有欄位都是必填的', 'error')
            return render_template('new_post.html')
        
        # 儲存到資料庫
        conn = get_db()
        cursor = conn.cursor()
        
        cursor.execute(
            'INSERT INTO posts (title, content, author) VALUES (?, ?, ?)',
            (title, content, author)
        )
        
        conn.commit()
        post_id = cursor.lastrowid
        conn.close()
        
        flash('文章發布成功', 'success')
        return redirect(url_for('post_detail', post_id=post_id))
    
    return render_template('new_post.html')


@app.route('/edit/', methods=['GET', 'POST'])
def edit_post(post_id):
    """編輯文章"""
    conn = get_db()
    cursor = conn.cursor()
    
    if request.method == 'POST':
        title = request.form.get('title', '').strip()
        content = request.form.get('content', '').strip()
        
        if not title or not content:
            flash('標題和內容不能為空', 'error')
            return redirect(url_for('edit_post', post_id=post_id))
        
        cursor.execute(
            'UPDATE posts SET title = ?, content = ? WHERE id = ?',
            (title, content, post_id)
        )
        
        conn.commit()
        conn.close()
        
        flash('文章更新成功', 'success')
        return redirect(url_for('post_detail', post_id=post_id))
    
    # GET：顯示編輯表單
    cursor.execute('SELECT * FROM posts WHERE id = ?', (post_id,))
    post = cursor.fetchone()
    conn.close()
    
    if not post:
        flash('文章不存在', 'error')
        return redirect(url_for('index'))
    
    return render_template('edit_post.html', post=post)


@app.route('/delete/')
def delete_post(post_id):
    """刪除文章"""
    conn = get_db()
    cursor = conn.cursor()
    
    cursor.execute('DELETE FROM posts WHERE id = ?', (post_id,))
    
    conn.commit()
    conn.close()
    
    flash('文章已刪除', 'success')
    return redirect(url_for('index'))


if __name__ == '__main__':
    app.run(debug=True)

9. RESTful API 開發

基本 REST API

# api.py
from flask import Flask, jsonify, request

app = Flask(__name__)

# 模擬資料庫
users = [
    {'id': 1, 'name': 'Alice', 'email': 'alice@example.com'},
    {'id': 2, 'name': 'Bob', 'email': 'bob@example.com'},
]

next_id = 3


# GET /api/users - 取得所有使用者
@app.route('/api/users', methods=['GET'])
def get_users():
    return jsonify({'users': users})


# GET /api/users/ - 取得單一使用者
@app.route('/api/users/', methods=['GET'])
def get_user(user_id):
    user = next((u for u in users if u['id'] == user_id), None)
    
    if user:
        return jsonify(user)
    else:
        return jsonify({'error': 'User not found'}), 404


# POST /api/users - 新增使用者
@app.route('/api/users', methods=['POST'])
def create_user():
    global next_id
    
    data = request.get_json()
    
    # 驗證
    if not data or 'name' not in data or 'email' not in data:
        return jsonify({'error': 'Missing required fields'}), 400
    
    # 新增使用者
    user = {
        'id': next_id,
        'name': data['name'],
        'email': data['email']
    }
    
    users.append(user)
    next_id += 1
    
    return jsonify(user), 201


# PUT /api/users/ - 更新使用者
@app.route('/api/users/', methods=['PUT'])
def update_user(user_id):
    user = next((u for u in users if u['id'] == user_id), None)
    
    if not user:
        return jsonify({'error': 'User not found'}), 404
    
    data = request.get_json()
    
    # 更新欄位
    if 'name' in data:
        user['name'] = data['name']
    if 'email' in data:
        user['email'] = data['email']
    
    return jsonify(user)


# DELETE /api/users/ - 刪除使用者
@app.route('/api/users/', methods=['DELETE'])
def delete_user(user_id):
    global users
    
    user = next((u for u in users if u['id'] == user_id), None)
    
    if not user:
        return jsonify({'error': 'User not found'}), 404
    
    users = [u for u in users if u['id'] != user_id]
    
    return jsonify({'message': 'User deleted'}), 200


if __name__ == '__main__':
    app.run(debug=True)
